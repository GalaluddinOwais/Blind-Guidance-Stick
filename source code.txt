void servo(String s){
int highTime = 0;



 if(s == "left")  highTime =2170;
else if(s == "slightlyLeft")   highTime =1770;
else if(s == "infront" )  highTime =1180;
else if(s == "slightlyRight")   highTime =760;
else if(s ==  "right"  ) highTime =310;


 if(s == "left" || s == "slightlyLeft" || s == "infront"  || s == "slightlyRight"  || s == "right" ) {
digitalWrite(6,HIGH);
delayMicroseconds(highTime);
 digitalWrite(6,LOW); 
}
}
